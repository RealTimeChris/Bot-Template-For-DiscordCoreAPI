# Set this value...
set(VCPKG_ROOT_DIR "C:/Vcpkg")
# OR this value...
set(CMAKE_CONFIG_FILE_DIR "")
# AND these values.
set(FFMPEG_INCLUDE_DIR "")
set(FFMPEG_RELEASE_ROOT "")
set(FFMPEG_DEBUG_ROOT "")
set(NLOHMANN_INCLUDE_DIR "")
set(OPENSSL_INCLUDE_DIR "")
set(OPENSSL_RELEASE_ROOT "")
set(OPENSSL_DEBUG_ROOT "")
set(OPUS_INCLUDE_DIR "")
set(OPUS_RELEASE_ROOT "")
set(OPUS_DEBUG_ROOT "")
set(SODIUM_INCLUDE_DIR "")
set(SODIUM_RELEASE_ROOT "")
set(SODIUM_DEBUG_ROOT "")

cmake_minimum_required(VERSION 3.20)
set(PROJECT_NAME "DiscordCoreAPI-Bot")
set(PRODUCT_VERSION "1.0.0.9")
set(DESCRIPTION "A Discord bot, written in C++, using the DiscordCoreAPI Discord bot library.")
project(
	"${PROJECT_NAME}"
	VERSION "${PRODUCT_VERSION}"
	LANGUAGES CXX
	HOMEPAGE_URL "https://discordcoreapi.com"
	DESCRIPTION "${DESCRIPTION}"
)
set(CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/CMake;${CMAKE_SOURCE_DIR}/CMake/Find-Scripts") 
set(CMAKE_CONFIGURATION_TYPES "${CMAKE_BUILD_TYPE}")

if(NOT EXISTS "${VCPKG_ROOT_DIR}" AND EXISTS "${CMAKE_CONFIG_FILE_DIR}")
	set(DiscordCoreAPI_DIR "${CMAKE_CONFIG_FILE_DIR}")
	if (NOT EXISTS "${FFMPEG_INCLUDE_DIR}" OR NOT EXISTS "${FFMPEG_RELEASE_ROOT}" OR NOT EXISTS "${FFMPEG_DEBUG_ROOT}"
		OR NOT EXISTS "${NLOHMANN_INCLUDE_DIR}" 
		OR NOT EXISTS "${OPENSSL_INCLUDE_DIR}" OR NOT EXISTS "${OPENSSL_RELEASE_ROOT}" OR NOT EXISTS "${OPENSSL_DEBUG_ROOT}"
		OR NOT EXISTS "${OPUS_INCLUDE_DIR}" OR NOT EXISTS "${OPUS_RELEASE_ROOT}" OR NOT EXISTS "${OPUS_DEBUG_ROOT}"
		OR NOT EXISTS "${SODIUM_INCLUDE_DIR}" OR NOT EXISTS "${SODIUM_RELEASE_ROOT}" OR NOT EXISTS "${SODIUM_DEBUG_ROOT}")
		
		message(FATAL_ERROR "Sorry - but please set the <DEPENDENCY>_DEBUG_ROOT, <DEPENDENCY>_RELEASE_ROOT AND <DEPENDENCY>_INCLUDE_DIR values in CMakeLists.txt!")
	endif()	
elseif(EXISTS "${VCPKG_ROOT_DIR}")
	include("${VCPKG_ROOT_DIR}/scripts/buildsystems/vcpkg.cmake")
	string(TOLOWER "${CMAKE_HOST_SYSTEM_NAME}" OS_NAME)
	set(VCPKG_INSTALLED_DIR "${VCPKG_ROOT_DIR}/installed/x64-${OS_NAME}")
	set(VCPKG_INCLUDE_DIR "${VCPKG_INSTALLED_DIR}/include")
	set(VCPKG_RELEASE_ROOT "${VCPKG_INSTALLED_DIR}/lib")
	set(VCPKG_DEBUG_ROOT "${VCPKG_INSTALLED_DIR}/debug/lib")
	set(FFMPEG_INCLUDE_DIR "${VCPKG_INCLUDE_DIR}")
	set(FFMPEG_RELEASE_ROOT "${VCPKG_RELEASE_ROOT}")
	set(FFMPEG_DEBUG_ROOT "${VCPKG_DEBUG_ROOT}")
	set(NLOHMANN_INCLUDE_DIR "${VCPKG_INCLUDE_DIR}")
	set(OPENSSL_INCLUDE_DIR "${VCPKG_INCLUDE_DIR}")
	set(OPENSSL_RELEASE_ROOT "${VCPKG_RELEASE_ROOT}")
	set(OPENSSL_DEBUG_ROOT "${VCPKG_DEBUG_ROOT}")
	set(OPUS_INCLUDE_DIR "${VCPKG_INCLUDE_DIR}")
	set(OPUS_RELEASE_ROOT "${VCPKG_RELEASE_ROOT}")
	set(OPUS_DEBUG_ROOT "${VCPKG_DEBUG_ROOT}")
	set(SODIUM_INCLUDE_DIR "${VCPKG_INCLUDE_DIR}")
	set(SODIUM_RELEASE_ROOT "${VCPKG_RELEASE_ROOT}")
	set(SODIUM_DEBUG_ROOT "${VCPKG_DEBUG_ROOT}")
else()
	message(FATAL_ERROR "Please, set the VCPKG base directory, or the location of the DiscordCoreAPIConfig.cmake file!")
endif()

find_package(FFMPEG MODULE REQUIRED)
find_package(NLOHMANN MODULE REQUIRED)
find_package(OPENSSL MODULE REQUIRED)
find_package(OPUS MODULE REQUIRED)
find_package(SODIUM MODULE REQUIRED)
find_package(DiscordCoreAPI CONFIG REQUIRED)

add_subdirectory(Executable)
